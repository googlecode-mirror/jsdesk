Creating A Module

1. Green font shows code that utilizes methods of the desktop object.
2. Red font is code that is required for a module in this desktop environment.

Code:
Ext.namespace("MyDesktop");

MyDesktop.BogusModule = Ext.extend(Ext.app.Module, {

	appType : 'bogus',
	id : 'bogus-win',
	
	init : function(){
		this.launcher = {
			text: 'Bogus Window',
			iconCls: 'bogus',
			handler: this.createWindow,
			scope: this
		}
	},

	createWindow : function(){
		var desktop = this.app.getDesktop();
		var win = desktop.getWindow('bogus-win');
        
		var winWidth = desktop.getWinWidth() / 1.1; 
		var winHeight = desktop.getWinHeight() / 1.1;

		if(!win){
			win = desktop.createWindow({
				id: 'bogus-win',
				title: 'Bogus Window',
				width: winWidth,
				height: winHeight,
				html: 'Something useful would be in here.',
				iconCls: 'bogus',
				shim: false,
				animCollapse: false,
				constrainHeader: true,
				x: desktop.getWinX(winWidth),
				y: desktop.getWinY(winHeight),
			});
		}

		win.show();
	}
});
